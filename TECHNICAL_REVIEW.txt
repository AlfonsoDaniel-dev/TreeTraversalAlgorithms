package Node

type Node struct {
	Id       int
	Parent   *Node
	children []*Node
	// Physics properties
	PositionX float64
	PositionY float64
	VelocityX float64
	VelocityY float64
	ForceX    float64
	ForceY    float64
	Mass      float64
}

func (n *Node) GetChildren() []*Node {
	return n.children
}

func (n *Node) AddChild(child *Node) error {
	if child == nil {
		return nil
	}
	n.children = append(n.children, child)
	return nil
}

func (n *Node) GetId() int {
	return n.Id
}

func (n *Node) GetParent() *Node {
	return n.Parent
}

func (n *Node) PopChildren(id int) {
	for i, child := range n.children {
		if child.GetId() == id {
			n.children = append(n.children[:i], n.children[i+1:]...)
			return
		}
		continue
	}
}

func NewNode(id int, parent *Node) *Node {
	return &Node{
		Id:        id,
		Parent:    parent,
		children:  make([]*Node, 0),
		PositionX: 0,
		PositionY: 0,
		VelocityX: 0,
		VelocityY: 0,
		ForceX:    0,
		ForceY:    0,
		Mass:      1.0,
	}
}

package Tree

import (
	"errors"
	"fmt"
	"github.com/AlfonsoDaniel-dev/TreeTraversal/src/Node"
)

type Tree struct {
	LastNodeId int
	Root       *Node.Node
	Nodes      map[int]*Node.Node
}

func NewTree() *Tree {
	root := Node.NewNode(0, nil)
	t := &Tree{
		Root:       root,
		LastNodeId: 0,
		Nodes:      make(map[int]*Node.Node),
	}
	t.Nodes[t.LastNodeId] = root
	return t
}

func (t *Tree) AddNodeFromRoot() error {
	if t.Root == nil {
		errStr := fmt.Sprintf("not root on the tree")
		return errors.New(errStr)
	}
	nodeId := t.LastNodeId + 1
	node := Node.NewNode(nodeId, t.Root)
	t.Root.AddChild(node)
	t.Nodes[nodeId] = node
	t.LastNodeId++
	return nil
}

func (t *Tree) AddNode(NodeId int) error {
	node, ok := t.Nodes[NodeId]
	if !ok {
		errStr := fmt.Sprintf("error: no Node with the id: %d found", NodeId)
		return errors.New(errStr)
	}
	nodeId := t.LastNodeId + 1
	newNode := Node.NewNode(nodeId, node)
	node.AddChild(newNode)
	t.Nodes[nodeId] = newNode
	t.LastNodeId++
	return nil
}

KEY FEATURES:
- O(1) node lookup via map
- Tree construction with automatic ID management
- File-based and programmatic tree generation support

package Tree

import (
	"errors"
	"github.com/AlfonsoDaniel-dev/TreeTraversal/src/Node"
)

type BfsState struct {
	CurrentNode *Node.Node
	Visited     []*Node.Node
	queue       []*Node.Node
	Unvisited   []*Node.Node
}

func (s BfsState) GetCurrent() *Node.Node {
	return s.CurrentNode
}

func (s BfsState) GetVisited() []*Node.Node {
	return s.Visited
}

func (s BfsState) GetFrontier() []*Node.Node {
	return s.queue
}

func (s BfsState) GetUnseen() []*Node.Node {
	return s.Unvisited
}

func (t *Tree) TraversalBfsSteps() ([]TraversalStep, error) {
	if t.Root == nil {
		return nil, errors.New("root is nil")
	}

	var history []TraversalStep
	queue := []*Node.Node{t.Root}
	discovered := make(map[int]bool)
	discovered[t.Root.Id] = true

	var visited []*Node.Node
	stepId := 0

	for len(queue) > 0 {
		actual := queue[0]
		queue = queue[1:]

		for _, child := range actual.GetChildren() {
			if !discovered[child.Id] {
				discovered[child.Id] = true
				queue = append(queue, child)
			}
		}

		var undiscovered []*Node.Node
		for id, node := range t.Nodes {
			if !discovered[id] {
				undiscovered = append(undiscovered, node)
			}
		}

		queueSnapshot := make([]*Node.Node, len(queue))
		copy(queueSnapshot, queue)

		visitedSnapshot := make([]*Node.Node, len(visited))
		copy(visitedSnapshot, visited)

		step := BfsState{
			CurrentNode: actual,
			Visited:     visitedSnapshot,
			queue:       queueSnapshot,
			Unvisited:   undiscovered,
		}

		history = append(history, TraversalStep{
			Id:    stepId,
			State: step,
		})

		visited = append(visited, actual)
		stepId++
	}

	return history, nil
}

KEY FEATURES:
- Records state at each traversal step
- Maintains queue, visited, and unvisited sets
- Implements TraversalState interface for polymorphism
- Time Complexity: O(V + E)
- Space Complexity: O(V) for state recording

ALGORITHM PROCESS:
1. Initialize queue with root node
2. For each node in queue:
   a. Record current state (snapshot)
   b. Process all unvisited children
   c. Mark children as discovered
   d. Add children to queue
   e. Append step to history
3. Return complete step history

package Tree

import (
	"errors"
	"github.com/AlfonsoDaniel-dev/TreeTraversal/src/Node"
)

type DfsState struct {
	CurrentNode *Node.Node
	Visited     []*Node.Node
	stack       []*Node.Node
	Unvisited   []*Node.Node
}

func (s DfsState) GetCurrent() *Node.Node    { return s.CurrentNode }
func (s DfsState) GetVisited() []*Node.Node  { return s.Visited }
func (s DfsState) GetFrontier() []*Node.Node { return s.stack }
func (s DfsState) GetUnseen() []*Node.Node   { return s.Unvisited }

func (t *Tree) TraversalDfsSteps() ([]TraversalStep, error) {
	if t.Root == nil {
		return nil, errors.New("no root in the tree")
	}

	var history []TraversalStep
	stack := []*Node.Node{t.Root}

	discovered := make(map[int]bool)
	discovered[t.Root.Id] = true

	var visited []*Node.Node
	stepId := 0

	for len(stack) > 0 {
		n := len(stack) - 1
		actual := stack[n]
		stack = stack[:n]

		children := actual.GetChildren()

		for i := len(children) - 1; i >= 0; i-- {
			child := children[i]
			if !discovered[child.Id] {
				discovered[child.Id] = true
				stack = append(stack, child)
			}
		}

		var undiscoverd []*Node.Node
		for id, node := range t.Nodes {
			if !discovered[id] {
				undiscoverd = append(undiscoverd, node)
			}
		}

		stackSnapshot := make([]*Node.Node, len(stack))
		copy(stackSnapshot, stack)

		visitedSnapshot := make([]*Node.Node, len(visited))
		copy(visitedSnapshot, visited)

		state := DfsState{
			CurrentNode: actual,
			Visited:     visitedSnapshot,
			stack:       stackSnapshot,
			Unvisited:   undiscoverd,
		}

		history = append(history, TraversalStep{
			Id:    stepId,
			State: state,
		})

		visited = append(visited, actual)
		stepId++
	}

	return history, nil
}

KEY FEATURES:
- Records state at each traversal step
- Maintains stack, visited, and unvisited sets
- Implements TraversalState interface for polymorphism
- Time Complexity: O(V + E)
- Space Complexity: O(V) for state recording
- Processes children in reverse order for natural traversal

ALGORITHM PROCESS:
1. Initialize stack with root node
2. For each node in stack:
   a. Pop node from stack
   b. Record current state (snapshot)
   c. Process children in reverse (for proper stack behavior)
   d. Mark unvisited children as discovered
   e. Push children to stack
   f. Append step to history
3. Return complete step history

PHYSICS SIMULATION (updatePhysics method):
================================================================================
The physics engine applies four forces to each node:

1. COULOMB REPULSION (All node pairs)
   - Prevents overlap between nodes
   - Formula: F = k / (distance² + epsilon)
   - Const repulsion = 4000.0

2. HOOK SPRING ATTRACTION (Connected nodes)
   - Maintains parent-child relationships
   - Formula: F = k * (distance - equilibrium_length)
   - Const springLen = 100.0, springK = 0.06

3. GRAVITY (Center attraction)
   - Pulls nodes toward screen center
   - Formula: F = gravity_force
   - Const gravity = 0.015
   - Target: screen center

4. VELOCITY DAMPING (Friction)
   - Reduces velocity each frame
   - Formula: velocity *= friction_coefficient
   - Const friction = 0.82
   - Also enforced: maxSpeed = 20.0

INTEGRATION:
- Update velocities: v += a (a = F/mass, mass = 1.0)
- Apply friction: v *= friction
- Clamp max speed: |v| <= maxSpeed
- Update positions: x += v
- Boundary enforcement (optional): keep nodes on-screen

CODE EXCERPT (Physics Engine):

const (
	repulsion  = 4000.0
	springLen  = 100.0
	springK    = 0.06
	gravity    = 0.015
	friction   = 0.82
	maxSpeed   = 20.0
)

// Coulomb Repulsion
for id1, n1 := range g.Nodes {
	for id2, n2 := range g.Nodes {
		if id1 == id2 {
			continue
		}
		dx, dy := n2.X-n1.X, n2.Y-n1.Y
		if math.Abs(dx) > 350 || math.Abs(dy) > 350 {
			continue
		}
		distSq := dx*dx + dy*dy + 0.1
		f := repulsion / distSq
		dist := math.Sqrt(distSq)
		n1.VX -= (dx / dist) * f
		n1.VY -= (dy / dist) * f
	}
}

// Hook Spring Attraction
for _, edge := range g.Edges {
	dx, dy := edge.To.X-edge.From.X, edge.To.Y-edge.From.Y
	dist := math.Sqrt(dx*dx + dy*dy)
	if dist > 1 {
		f := (dist - springLen) * springK
		edge.From.VX += (dx / dist) * f
		edge.From.VY += (dy / dist) * f
		edge.To.VX -= (dx / dist) * f
		edge.To.VY -= (dy / dist) * f
	}
}

// Gravity + Damping
cx, cy := float64(g.ScreenWidth/2), float64(g.ScreenHeight/2)
for _, n := range g.Nodes {
	if n == g.draggedNode {
		continue
	}
	n.VX += (cx - n.X) * gravity
	n.VY += (cy - n.Y) * gravity
	n.VX, n.VY = n.VX*friction, n.VY*friction

	// Speed clamping
	if n.VX > maxSpeed {
		n.VX = maxSpeed
	} else if n.VX < -maxSpeed {
		n.VX = -maxSpeed
	}
	if n.VY > maxSpeed {
		n.VY = maxSpeed
	} else if n.VY < -maxSpeed {
		n.VY = -maxSpeed
	}

	n.X, n.Y = n.X+n.VX, n.Y+n.VY
}

type ArbolJSON struct {
	Nombre string `json:"nombre"`
	Padres []int  `json:"padres"`
}

func GenerarArbolAleatorio(numNodos int) *Tree.Tree {
	t := Tree.NewTree()
	for i := 1; i < numNodos; i++ {
		t.AddNode(rand.Intn(i))
	}
	return t
}

func GenerarArbolProfundo(niveles int) *Tree.Tree {
	t := Tree.NewTree()
	for i := 0; i < niveles; i++ {
		t.AddNode(i)
	}
	return t
}

func main() {
	ebiten.SetWindowTitle("Visualizador BFS/DFS - Ebitengine")
	ebiten.SetWindowSize(1920, 1080)
	ebiten.SetWindowResizingMode(ebiten.WindowResizingModeEnabled)

	var catalogTrees []*Tree.Tree
	var catalogNames []string

	// 1. Load from JSON (arboles.json)
	data, err := os.ReadFile("./cmd/arboles.json")
	if err == nil {
		var catalogo []ArbolJSON
		if err := json.Unmarshal(data, &catalogo); err == nil {
			for _, arbolData := range catalogo {
				t := Tree.NewTree()
				for _, padreID := range arbolData.Padres {
					t.AddNode(padreID)
				}
				catalogTrees = append(catalogTrees, t)
				catalogNames = append(catalogNames, arbolData.Nombre)
			}
		} else {
			log.Println("Error parsing JSON:", err)
		}
	} else {
		log.Println("arboles.json not found, continuing with local generators.")
	}

	// 2. Generate massive trees programmatically
	catalogTrees = append(catalogTrees, GenerarArbolAleatorio(150))
	catalogNames = append(catalogNames, "Random Monster (150 Nodes)")

	catalogTrees = append(catalogTrees, GenerarArbolProfundo(50))
	catalogNames = append(catalogNames, "Deep Whip (50 Nodes)")

	// 3. Safety fallback if no trees
	if len(catalogTrees) == 0 {
		catalogTrees = append(catalogTrees, Tree.NewTree())
		catalogNames = append(catalogNames, "Empty Tree")
	}

	// 4. Initialize game
	bgColor := color.RGBA{R: 15, G: 15, B: 25, A: 255}
	game := Game.NewGame(bgColor, 1280, 720, catalogTrees[0])

	game.CatalogTrees = catalogTrees
	game.CatalogNames = catalogNames
	game.CurrentTreeIndex = 0

	log.Println("System initialized. Catalog loaded:")
	for i, name := range catalogNames {
		log.Printf(" [%d] %s", i+1, name)
	}

	if err := ebiten.RunGame(game); err != nil {
		log.Fatal("Error running game:", err)
	}
}

PHYSICS ENGINE STRUCTURE:
======================
All forces calculated per frame in updatePhysics():

Force Calculation Pipeline:
1. Coulomb repulsion (O(n²))
2. Hook spring attraction (O(e) where e = # edges)
3. Center gravity (O(n))
4. Friction damping (O(n))
5. Integration (O(n))

Total Complexity: O(n² + e) ≈ O(n²) for typical trees

================================================================================
                     PHYSICS FORMULAS REFERENCE
================================================================================

1. COULOMB REPULSION
   ─────────────────
   Electric force between charged particles.
   Applied to all node pairs to prevent overlap.

   Physics Formula:
   F = k_e * (q₁ * q₂) / r²

   Implementation:
   - k_e = repulsion constant (4000.0)
   - q₁, q₂ = unit charges (1.0)
   - r = distance between nodes

   Code:
   distSq := dx*dx + dy*dy + 0.1
   f := repulsion / distSq
   dist := math.Sqrt(distSq)
   n1.VX -= (dx / dist) * f
   n1.VY -= (dy / dist) * f

   Effect: Nodes push each other apart

2. HOOK SPRING ATTRACTION
  ──────────────────────
   Spring force between connected nodes.
   Maintains parent-child physical relationships.

   Physics Formula:
   F = -k * (x - L₀)

   Where:
   - k = spring constant (0.06)
   - x = current distance
   - L₀ = natural length (100.0)

   Code:
   f := (dist - springLen) * springK
   edge.From.VX += (dx / dist) * f
   edge.To.VX -= (dx / dist) * f

   Effect: Connected nodes attract/repel to maintain distance

3. GRAVITY (Center Attraction)
   ────────────────────────────
   Continuous downward force on all nodes.
   Keeps tree structure centered on screen.

   Physics Formula:
   F = g * m (toward center)

   Where:
   - g = gravity constant (0.015)
   - m = mass (1.0)
   - Direction: toward (screenWidth/2, screenHeight/2)

   Code:
   cx, cy := float64(g.ScreenWidth/2), float64(g.ScreenHeight/2)
   n.VX += (cx - n.X) * gravity
   n.VY += (cy - n.Y) * gravity

   Effect: Attracts nodes toward screen center

4. FRICTION DAMPING
   ────────────────
   Velocity reduction each frame.
   Reduces oscillation and stabilizes layout.

   Physics Formula:
   v_new = v_old * D

   Where:
   - D = damping coefficient (0.82)
   - Range: 0.0 (instant stop) to 1.0 (no damping)

   Code:
   n.VX, n.VY = n.VX*friction, n.VY*friction

   Speed Clamping (prevents runaway nodes):
   if n.VX > maxSpeed:
       n.VX = maxSpeed

   Effect: Progressively slows nodes to equilibrium

5. VELOCITY INTEGRATION
   ────────────────────
   Updates position based on velocity each frame.
   Simple Euler integration method.

   Code:
   n.X, n.Y = n.X+n.VX, n.Y+n.VY

   Complete Update Cycle per Node:
   1. Calculate forces (F)
   2. Apply repulsion: VX -= f_rep_x, VY -= f_rep_y
   3. Apply spring: VX += f_spring_x, VY += f_spring_y
   4. Apply gravity: VX += g_x, VY += g_y
   5. Apply damping: V *= friction
   6. Clamp velocity: |V| <= maxSpeed
   7. Integrate: P += V

================================================================================
                      ALGORITHM COMPLEXITY
================================================================================

BFS ALGORITHM:
──────────────
Time Complexity: O(V + E)
  - V = number of nodes (vertices)
  - E = number of edges (parent-child relationships)
  - Each node visited once
  - Each edge examined once

Space Complexity: O(V)
  - Queue storage: O(h) where h = tree height
  - State snapshots: O(V log V) for all steps
  - Discovered map: O(V)

DFS ALGORITHM:
──────────────
Time Complexity: O(V + E)
  - Same as BFS for tree traversal
  - Order of traversal differs (LIFO vs FIFO)

Space Complexity: O(V)
  - Stack storage: O(h) where h = tree height
  - State snapshots: O(V log V) for all steps
  - Discovered map: O(V)

PHYSICS UPDATE:
───────────────
Time Complexity: O(n²) per frame
  - Coulomb repulsion: O(n²) all-pairs calculation
  - Hook attraction: O(e) where e ≈ n for trees
  - Gravity/Damping: O(n)
  - Overall: O(n²) dominated by repulsion

Space Complexity: O(n)
  - Node positions: O(n)
  - Node velocities: O(n)
  - Edge connections: O(e) ≈ O(n)

RENDERING:
──────────
Time Complexity: O(n + e)
  - n nodes rendered: O(n)
  - e edges rendered: O(e) ≈ O(n)
  - UI overlay: O(1)
  - Overall: O(n)

Space Complexity: O(n)
  - Screen buffer: O(w × h) constant
  - Per-node data: O(n)

TOTAL FRAME:
────────────
Per Frame Time Complexity: O(n²) dominated by physics
Output Frame Rate: 60 FPS target

Performance Budget per Frame:
- 16.67 ms for 60 FPS
- Physics: ~10 ms (O(n²))
- Rendering: ~3 ms (O(n))
- Input/Logic: ~2 ms
- Buffer: ~1.67 ms

Suitable Tree Size:
- n < 100: Very smooth (< 1 ms)
- n 100-300: Smooth (2-5 ms)
- n 300-1000: Acceptable (5-12 ms)
- n > 1000: May stutter (> 12 ms)
